    printf()
    ■ Exemplo:
        ◻ int i = 10;
        ◻ float r = 3.1514;
        ◻ Invertendo:

            int main()
            {
                int i = 10;
                float r = 7.1234;
                printf("Inteiro: %d, Real: %f \n", r, i);  //inteiro: -1610612736, Real: 0.000000
            }
            printf()

    scanf()
    ■ Lê qualquer tipo de dado primitivo;
    ■ Utiliza um texto formatado como máscara de leitura;
    ■ Utiliza o endereço das variáveis como parâmetro.
    #include <stdio.h>
    int main()
    {
        // declaracao das variaveis
        int numero;
        printf("Entre com um valor:");
        scanf("%d", &numero);
    }

    getchar()
    ■ Leitura de variáveis:
    ■ Função Get Char ou,
    ■ getchar():
        ◻ Lê um caractere (apenas 1) da entrada e o armazena em uma variável. Não lê números.
        ◻ Assinatura:
            ■ int getchar(void);
        ◻ A variável a ser lida deve receber o retorno da função.
        ◻ Pode ser utilizada apenas para bloquear o fluxo de
    execução do programa.
